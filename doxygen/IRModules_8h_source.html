58 /// Represents a Python MlirNamedAttr, carrying an optional owned name.
59 /// TODO: Refactor this and the C-API to be based on an Identifier owned
60 /// by the context so as to avoid ownership issues here.
61 class PyNamedAttribute {
62 public:
63  /// Constructs a PyNamedAttr that retains an owned name. This should be
64  /// used in any code that originates an MlirNamedAttribute from a python
65  /// string.
66  /// The lifetime of the PyNamedAttr must extend to the lifetime of the
67  /// passed attribute.
68  PyNamedAttribute(MlirAttribute attr, std::string ownedName);
69 
70  MlirNamedAttribute namedAttr;
71 
72 private:
73  // Since the MlirNamedAttr contains an internal pointer to the actual
74  // memory of the owned string, it must be heap allocated to remain valid.
75  // Otherwise, strings that fit within the small object optimization threshold
76  // will have their memory address change as the containing object is moved,
77  // resulting in an invalid aliased pointer.
78  std::unique_ptr<std::string> ownedName;
